# Phase 4: Chrome Extension -- App Domain Plan

## Objective

Build the server-side capture pipeline and parsing engine that receives raw HTML from the Chrome extension, compresses and stores it in the page cache, parses it into structured contact/company data using configurable CSS selectors via cheerio, upserts contacts with confidence-based field merging, and exposes extension-facing API endpoints for tasks, health, settings, registration, contact lookup, and message rendering.

```typescript
import * as cheerio from 'cheerio';
import { parserRegistry } from './parser-registry';
import { ParseResult, ExtractedField } from './types';
import { SelectorConfig, LinkedInPageType } from '@/types/selector-config';
import { decompressHtml } from '@/lib/capture/capture-store';
import { upsertContactFromParse } from './contact-upsert';
import { wsServer } from '@/lib/websocket/ws-server';
import { createParseCompleteEvent } from '@/lib/websocket/ws-events';

export class ParserEngine {
  /**
   * Main entry point: parse a cached page.
   *
   * Flow:
   *   1. Load compressed HTML from page_cache by captureId
   *   2. Decompress HTML
   *   3. Detect page type (already stored, but verify)
   *   4. Load active selector config for page type
   *   5. Get registered parser for page type
   *   6. Execute parser
   *   7. If profile/company: upsert contact/company record
   *   8. Update embeddings via ruvector_embed()
   *   9. Mark page_cache record as parsed
   *  10. Push PARSE_COMPLETE WebSocket event
   *  11. Return ParseResult
   */
  async parseCachedPage(captureId: string): Promise<ParseResult>;

  /**
   * Parse raw HTML directly (used by re-parse jobs).
   */
  async parseHtml(
    html: string,
    pageType: LinkedInPageType,
    url: string,
    captureId: string
  ): Promise<ParseResult>;

  /**
   * Load active selector config for a page type from DB.
   */
  private async loadSelectorConfig(pageType: LinkedInPageType): Promise<SelectorConfig>;

  /**
   * Apply a single selector chain against the DOM.
   * Tries selectors in order, returns first match with confidence
   * based on selector position (first = 1.0, second = 0.8, third = 0.6, etc.)
   */
  static applySelectorChain(
    $: CheerioAPI,
    chain: SelectorChain
  ): ExtractedField | null;

  /**
   * Apply heuristic rules as fallback extraction.
   */
  static applyHeuristics(
    $: CheerioAPI,
    fields: ExtractedField[],
    heuristics: HeuristicRule[]
  ): ExtractedField[];
}

export const parserEngine = new ParserEngine();
```

```typescript
import { CheerioAPI } from 'cheerio';
import { PageParser, ParseResult, ProfileParseData, ExperienceEntry, EducationEntry, ExtractedField } from '../types';
import { SelectorConfig, LinkedInPageType } from '@/types/selector-config';
import { ParserEngine } from '../parser-engine';

export class ProfileParser implements PageParser {
  readonly pageType: LinkedInPageType = 'PROFILE';
  readonly version = '1.0.0';

  parse($: CheerioAPI, config: SelectorConfig, url: string): Omit<ParseResult, 'captureId' | 'parseTimeMs'> {
    const fields: ExtractedField[] = [];
    const errors: string[] = [];

    // Extract each field using selector chains from config
    const nameField = ParserEngine.applySelectorChain($, config.selectors.name);
    if (nameField) fields.push(nameField);

    const headlineField = ParserEngine.applySelectorChain($, config.selectors.headline);
    if (headlineField) fields.push(headlineField);

    // ... (location, about, connectionsCount, skills)

    // Complex extraction: experience entries
    const experienceEntries = this.parseExperience($, config);

    // Complex extraction: education entries
    const educationEntries = this.parseEducation($, config);

    // Apply heuristic fallbacks
    const heuristicFields = ParserEngine.applyHeuristics($, fields, config.heuristics);
    fields.push(...heuristicFields);

    const data: ProfileParseData = {
      name: this.getFieldValue(fields, 'name') as string | null,
      headline: this.getFieldValue(fields, 'headline') as string | null,
      location: this.getFieldValue(fields, 'location') as string | null,
      about: this.getFieldValue(fields, 'about') as string | null,
      connectionsCount: this.getFieldValue(fields, 'connectionsCount') as number | null,
      experience: experienceEntries,
      education: educationEntries,
      skills: (this.getFieldValue(fields, 'skills') as string[]) || [],
      profileImageUrl: this.getFieldValue(fields, 'profileImageUrl') as string | null,
    };

    const fieldsExtracted = fields.filter(f => f.value !== null).length;
    const fieldsAttempted = Object.keys(config.selectors).length;

    return {
      success: fieldsExtracted > 0,
      pageType: this.pageType,
      url,
      fields,
      data,
      fieldsExtracted,
      fieldsAttempted,
      overallConfidence: fields.length > 0
        ? fields.reduce((sum, f) => sum + f.confidence, 0) / fields.length
        : 0,
      parserVersion: this.version,
      selectorConfigVersion: config.version,
      errors,
    };
  }

  private parseExperience($: CheerioAPI, config: SelectorConfig): ExperienceEntry[];
  private parseEducation($: CheerioAPI, config: SelectorConfig): EducationEntry[];
  private getFieldValue(fields: ExtractedField[], fieldName: string): string | string[] | number | null;
}

// Register with parser registry
import { parserRegistry } from '../parser-registry';
```

```typescript
import { ProfileParseData, ParseResult, ExtractedField } from './types';

export interface UpsertResult {
  action: 'created' | 'updated' | 'unchanged';
  contactId: string;
  fieldsUpdated: string[];
  previousValues: Record<string, unknown>;
  newValues: Record<string, unknown>;
}

/**
 * Upsert contact from parsed profile data.
 *
 * Matching strategy:
 *   1. Match by LinkedIn profile URL (primary key equivalent)
 *   2. Normalize URL: strip tracking params (?miniProfile, ?trk, etc.)
 *   3. Ensure consistent format: https://www.linkedin.com/in/<username>
 *
 * Merge strategy (confidence-based):
 *   - For each field, compare new value confidence vs existing value confidence
 *   - Higher confidence wins (stored in contacts.field_confidences JSONB column)
 *   - If confidences are equal, prefer newer value
 *   - Never overwrite enrichment-sourced fields with lower-confidence parsed data
 *   - Track all field changes for audit trail
 *
 * Field confidence hierarchy:
 *   - Enrichment provider (PDL, Lusha): 0.95
 *   - First selector match: 0.90
 *   - Second selector match: 0.72
 *   - Third selector match: 0.54
 *   - Heuristic match: 0.50
 *   - CSV import: 0.40
 */
export async function upsertContactFromParse(
  parseResult: ParseResult,
  url: string
): Promise<UpsertResult>;

/**
 * Normalize LinkedIn URL for matching.
 */
export function normalizeLinkedInUrl(url: string): string;

/**
 * Compare field values with confidence and determine winner.
 */
export function resolveFieldConflict(
  existingValue: unknown,
  existingConfidence: number,
  newValue: unknown,
  newConfidence: number
): { value: unknown; confidence: number; updated: boolean };
```
